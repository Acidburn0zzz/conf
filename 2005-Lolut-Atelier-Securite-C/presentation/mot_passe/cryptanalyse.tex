\section{Cryptanalyse}

\subsection{Introduction}

\begin{frame}
  \frametitle{Introduction}

  \begin{columns}
    \begin{column}{1.0\textwidth}<1->
      \begin{itemize}
      \item<1-> La cryptanalyse est la science qui étudie les algorithmes
      de chiffrement et de signature.
      \item<2-> Le chiffrement et les signatures utilisent largement les
      mathématiques.
      \item<3-> Les algorithmes sont torturés pour qu'ils montrent leurs
      faiblesses. Lorsqu'un algorithme est "cassé", un nouveau va le
      remplacer.
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}
