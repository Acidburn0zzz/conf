\section{Enjeux et risques}

\subsection{Enjeux}

\begin{frame}
  \frametitle{Enjeux de la sécurité}

  \begin{columns}
    \begin{column}{0.7\textwidth}<1->
      \begin{itemize}
      \item<1-> Le ver Code Red a causé des dégats de l'ordre de milliards de dollar US (!). 
      \item<2-> Vol de données (ex: numéros de carte de crédit,
      espionnage industriel).
      \item<3-> Des bris de sécurité peuvent causer du tort à la vie privée
      et à la confidentialité.
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\subsection{Risques}

\begin{frame}
  \frametitle{Quelques risques liés à la sécurité}

  \begin{columns}
    \begin{column}{0.7\textwidth}<1->
      \begin{itemize}
      \item<1-> Corruption et perte des données (ver, virus)
      \item<2-> Intrusion dans le système (trojan, exploit, rootkit)
      \item<3-> Envoi d'informations à notre insu (spyware)
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}
