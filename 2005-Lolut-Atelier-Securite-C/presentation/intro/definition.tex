\section{Définitions de la sécurité}

\subsection{Sécurité = disponibilité ?}

\begin{frame}
  \frametitle{Sécurité = disponibilité}

  \begin{columns}
    \begin{column}{0.7\textwidth}<1->
      \begin{itemize}
      \item « La sécurité informatique a pour but de garder un
      ordinateur opérationnel dans n'importe quelle condition »
      \item Tolérance aux pannes matérielles
      \item Tolérance aux pannes logicielles
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\subsection{Sécurité = confidentialité ?}

\begin{frame}
  \frametitle{Sécurité = confidentialité}

  \begin{columns}
    \begin{column}{0.7\textwidth}<1->
      \begin{itemize}
      \item<1-> « La sécurité informatique a pour but d'assurer que les
      ressources ne sont accessibles qu'aux personnes autorisées »
      \item<2-> Gestion fine des droits 
      \item<3-> Système d'authentification robuste
      \item<4-> Peut utiliser le chiffrement des données
      et la signature des documents
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\subsection{Sécurité = intégrité ?}

\begin{frame}
  \frametitle{Sécurité = intégrité}

  \begin{columns}
    \begin{column}{0.7\textwidth}<1->
      \begin{itemize}
      \item<1-> « La sécurité informatique a pour but d'assurer que les
      données ne sont pas modifées, altérées ou détruites »
      \item<2-> Intégrité lors du stockage et du transfert des données
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\subsection{Et finalement ?}

\begin{frame}
  \frametitle{La sécurité est la combinaison des trois}

  \begin{columns}
    \begin{column}{0.7\textwidth}<1->
      \begin{itemize}
      \item<1-> Disponibilité
      \item<2-> Confidentialité
      \item<3-> Intégrité
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}
