\section{Principe du maillon faible}

\subsection{Présentation du principe}

\begin{frame}
  \frametitle{Principe du maillon faible}

  \begin{columns}
    \begin{column}{0.7\textwidth}<1->
      \begin{itemize}
      \item<1-> La sécurité d'un système est celle de l'élément le plus faible. 
      \item<2-> Il faut connaitre tous les éléments du système, ne rien négliger.
      \item<3-> Rien ne sert d'installer une porte blindée sur un mur en carton. 
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}
  \frametitle{Exemple : Faille dans Linux}
  \framesubtitle{Extrait de l'exploit chown}

  \begin{center}
     chown(argv[1], -1, getgid());

     \begin{itemize}
       \item argv[1] : Nom du fichier
       \item -1 : Propriétaire du fichier, -1 signifie "pas de changement"
       \item getgid() : Groupe du fichier, utilise le groupe de l'utilisateur
       ayant lancé le programme.
     \end{itemize}
  \end{center}
\end{frame}
